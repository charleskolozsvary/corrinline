\documentclass{article}
\usepackage{luacode}

\begin{luacode*}
-- Open output file
local outfile = io.open("boxes.txt", "w")
local node_counter = 0

-- Function to record position
function record_position(id, page, nodetype, width_sp, height_sp, depth_sp)
    local x_sp = tex.sp("\\the\\pdflastxpos sp")
    local y_sp = tex.sp("\\the\\pdflastypos sp")
    
    local x = x_sp / 65536
    local y = y_sp / 65536
    local width = width_sp / 65536
    local height = height_sp / 65536
    local depth = depth_sp / 65536
    
    local x_start = x
    local x_end = x + width
    local y_start = y - height
    local y_end = y + depth
    
    outfile:write(string.format(
        "id=%d, page=%d, type=%s, x=[%.2f, %.2f], y=[%.2f, %.2f]\n",
        id, page, nodetype, x_start, x_end, y_start, y_end
    ))
end

-- Insert position markers
local function instrument_nodes(head)
    for n in node.traverse(head) do
        local id = node.type(n.id)
        
        if (id == "hlist" or id == "vlist" or id == "glyph" or id == "rule") and n.width then
            local width = n.width
            local height = n.height or 0
            local depth = n.depth or 0
            local current_id = node_counter
            
            -- Insert via tex.sprint
            tex.sprint("\\pdfsavepos\\latelua{record_position(" .. current_id .. "," .. 
                       tex.count["c@page"] .. ",'" .. id .. "'," .. 
                       width .. "," .. height .. "," .. depth .. ")}")
            
            node_counter = node_counter + 1
        end
        
        if (id == "hlist" or id == "vlist") and n.head then
            n.head = instrument_nodes(n.head)
        end
    end
    
    return head
end

luatexbase.add_to_callback("pre_linebreak_filter", instrument_nodes, "instrument_positions")

luatexbase.add_to_callback("stop_run", function()
    if outfile then
        outfile:close()
        texio.write_nl("Box information written to boxes.txt")
    end
end, "close_file")
\end{luacode*}
